        "function backtracking_line_search_unconstrained(f, grad, x, dx; alpha = 0.4, beta = 0.5)\n",
        "    t = 1.0\n",
        "    fx = f(x)\n",
        "    gx = grad(x)\n",
        "    while f(x .+ t .* dx) > fx + alpha * t * dot(gx, dx)\n",
        "        t *= beta\n",
        "    end\n",
        "    return t\n",
        "end\n",
        "\n",
        "function gradient_descent_armijo(f, grad, x0; alpha = 0.4, beta = 0.5, tol = 1e-8, maxiter = 10_000)\n",
        "    x = copy(x0)\n",
        "    trajectory = [copy(x)]\n",
        "    for _ in 1:maxiter\n",
        "        g = grad(x)\n",
        "        if norm(g) <= tol\n",
        "            return x, trajectory\n",
        "        end\n",
        "        dx = -g\n",
        "        t = backtracking_line_search_unconstrained(f, grad, x, dx; alpha = alpha, beta = beta)\n",
        "        x = x .+ t .* dx\n",
        "        push!(trajectory, copy(x))\n",
        "    end\n",
        "    return x, trajectory\n",
        "end\n",
        "\n",
        "function newton_fixed(grad, hess, x0; tol = 1e-8, maxiter = 100)\n",
        "    x = copy(x0)\n",
        "    trajectory = [copy(x)]\n",
        "    for _ in 1:maxiter\n",
        "        g = grad(x)\n",
        "        if norm(g) <= tol\n",
        "            return x, trajectory\n",
        "        end\n",
        "        x = x .- (hess(x) \\ g)\n",
        "        push!(trajectory, copy(x))\n",
        "    end\n",
        "    return x, trajectory\n",
        "end\n",

        "## Gradient Descent with Backtracking Line Search\n",
        "\n",
        "Given\n",
        "\\begin{align*}\n",
        "f(x_1, x_2, x_3) = x_3 \\log\\left(e^{x_1/x_3} + e^{x_2/x_3}\\right) + (x_3 - 2)^2 + e^{1/(x_1 + x_2)}\n",
        "\\end{align*}\n",
        "with domain $\\mathrm{dom}\\,f = \\{x \\in \\mathbb{R}^3 : x_1 + x_2 > 0,\\; x_3 > 0\\}$.\n",
        "\n",
        "Below we define the function, its gradient, the finite-difference Hessian, the domain-safe backtracking routine, and then run Gradient Descent, Newton and BFGS."

        "function in_domain(x)\n",
        "    return (x[1] + x[2] > 0.0) && (x[3] > 0.0)\n",
        "end\n",
        "\n",
        "function logsumexp2(a, b)\n",
        "    m = max(a, b)\n",
        "    return m + log(exp(a - m) + exp(b - m))\n",
        "end\n",
        "\n",
        "function softmax2(a, b)\n",
        "    m = max(a, b)\n",
        "    ea = exp(a - m)\n",
        "    eb = exp(b - m)\n",
        "    den = ea + eb\n",
        "    return ea / den, eb / den\n",
        "end\n",
        "\n",
        "function f_main(x)\n",
        "    in_domain(x) || error(\"Point outside dom f\")\n",
        "    x1, x2, x3 = x\n",
        "    s = x1 + x2\n",
        "    return x3 * logsumexp2(x1 / x3, x2 / x3) + (x3 - 2)^2 + exp(1 / s)\n",
        "end\n",
        "\n",
        "function grad_main(x)\n",
        "    in_domain(x) || error(\"Point outside dom f\")\n",
        "    x1, x2, x3 = x\n",
        "    s = x1 + x2\n",
        "    p1, p2 = softmax2(x1 / x3, x2 / x3)\n",
        "    lse = logsumexp2(x1 / x3, x2 / x3)\n",
        "    common = exp(1 / s) / s^2\n",
        "    return [\n",
        "        p1 - common,\n",
        "        p2 - common,\n",
        "        lse - (x1 * p1 + x2 * p2) / x3 + 2 * (x3 - 2)\n",
        "    ]\n",
        "end\n",
        "\n",
        "function finite_difference_hessian(grad, x; h = 1e-5)\n",
        "    n = length(x)\n",
        "    H = zeros(n, n)\n",
        "    g0 = grad(x)\n",
        "    for i in 1:n\n",
        "        ei = zeros(n)\n",
        "        ei[i] = h\n",
        "        H[:, i] = (grad(x .+ ei) .- g0) ./ h\n",
        "    end\n",
        "    return 0.5 .* (H + H')\n",
        "end\n",
        "\n",
        "alpha_bt = 0.4\n",
        "beta_bt = 0.5\n",
        "eps_stop = 1e-5\n",
        "h_fd = 1e-5\n",
        "x_start = [3.0, 4.0, 5.0]\n",
        "\n",
        "println(\"f(x_start)      = \", f_main(x_start))\n",
        "println(\"norm(grad_start)= \", norm(grad_main(x_start)))"

        "function ensure_domain_step(x, dx; beta = 0.5)\n",
        "    t = 1.0\n",
        "    while !in_domain(x .+ t .* dx)\n",
        "        t *= beta\n",
        "    end\n",
        "    return t\n",
        "end\n",
        "\n",
        "function backtracking_line_search(f, grad, x, dx; alpha = 0.4, beta = 0.5)\n",
        "    t = ensure_domain_step(x, dx; beta = beta)\n",
        "    fx = f(x)\n",
        "    gx = grad(x)\n",
        "    while f(x .+ t .* dx) > fx + alpha * t * dot(gx, dx)\n",
        "        t *= beta\n",
        "    end\n",
        "    return t\n",
        "end\n",
        "\n",
        "dx_gd0 = -grad_main(x_start)\n",
        "t_domain0 = ensure_domain_step(x_start, dx_gd0; beta = beta_bt)\n",

        "function gradient_descent_backtracking(f, grad, x0; alpha = 0.4, beta = 0.5, eps = 1e-5, maxiter = 10_000)\n",
        "    x = copy(x0)\n",
        "    history = [copy(x)]\n",

        "        g = grad(x)\n",
        "        if norm(g) <= eps\n",

        "        end\n",
        "        dx = -g\n",
        "        t = backtracking_line_search(f, grad, x, dx; alpha = alpha, beta = beta)\n",
        "        x = x .+ t .* dx\n",
        "        push!(history, copy(x))\n",
        "    end\n",
        "    error(\"Gradient descent did not converge in the maximum number of iterations.\")\n",

        "        end\n",
        "        p = -H * g\n",
        "        t = backtracking_line_search(f, grad, x, p; alpha = alpha, beta = beta)\n",

        "        y = g_new .- g\n",

        "        rho = ys <= ys_tol ? rho_cap : 1 / ys\n",
        "        I_n = Matrix{Float64}(I, n, n)\n",
        "        V = I_n - rho .* (s * y')\n",
        "        H = V * H * V' + rho .* (s * s')\n",
        "        x = x_new\n",
        "        push!(history, copy(x))\n",

        "    error(\"BFGS did not converge in the maximum number of iterations.\")\n",
